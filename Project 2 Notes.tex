\section{One-Sector Stochastic Growth Model}
\subsection*{Notes on optimality conditions}
We have the technology given by
	\begin{align*}
	z_{t+1} 	&= \exp(x_{t+1}) = \exp(\rho x_{t} + \epsilon_{t+1}) \\
	z_t 		&= \exp(x_t)	= \exp(\rho x_{t-1}+\epsilon_t).
	\end{align*}
We can solve the latter one for $\rho$:
	\begin{align*}
	\log z_t - \epsilon_t  = \rho x_{t-1} \Leftrightarrow \rho = \frac{\log z_t - \epsilon_t}{x_{t-1}},
	\end{align*}	
which we can plug into $z_{t+1}$ 
	\begin{align*}
	z_{t+1} = \exp\bigg( \frac{x_t}{x_{t-1}}(\log z_t - \epsilon_t) + \epsilon_{t+1} \bigg)
	\end{align*}
